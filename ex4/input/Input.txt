class Person
{
	int ID;
	int age := 18;

	int getAge()  { return age;   }
	int birthday(){ age := age+1; return age; }
}

class Student extends Person
{
	int grades;
	int getAverage()
	{
		int i := 0;
		int sum := 0;
		while (i<10)
		{
			sum := sum+grades;
			i := i+1;
		}
		return sum/10;
	}
}

void main()
{
	Student moish := new Student;
	moish.grades := 5;
	return;
}
